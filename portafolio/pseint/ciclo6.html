6. Se está creando una aplicación que va a estar conectada con un prototipo que permita almacenar contactos telefónicos en el dispositivo. Para ello cada contacto debe contener nombre completo, teléfono y organización. Se requiere que la aplicación permita añadir 3 contactos verificando que el número no esté almacenado, buscar contactos almacenados y eliminar contactos si el usuario lo requiere. Recuerde que el sistema debe terminar cuando el usuario así lo indique.

            Definir nombre,empresa,eliminarMas Como Caracter;

                    2:
                        Escribir " ";
                        Escribir "Los Usuarios Almacenados Son: ", contactos;
                        
                    3: 	Escribir "Desea Eliminar Contactos?";
                        Escribir "1: SI";

                        SiNo
                            Si eliminarContacto == 1 Entonces
                                contactos <- contactos -1; 
                                Escribir "El numero de contactos ahora es: ", contactos;
                                Si contactos == 0 Entonces
                                    Escribir "No Existen Contactos";
                                
                                FinSi
                                
                            FinSi
                        FinSi
                        
                        
                            
                        
                    De Otro Modo:
                        Escribir "Opcion No Valida";
                FinSegun
                
                
                Escribir " ";

                Escribir "1: SI";

                leer decision;

                    seguir <- Verdadero;

                    seguir <- Falso;

                Si contactos = 3 Entonces
                    Escribir "Limite de Contactos Alcanzados";
                    
                    
                FinSi
