SubProceso numbers()
	Definir datos, i Como Entero;
	Dimension datos[5];
	
	
	Para i <- 0 Hasta 4 Con Paso 1 Hacer
		Escribir "Escribe el Numero ", i+1;
		Leer datos[i];
	FinPara
	Escribir " ";
	Para i<- 0 Hasta 4 Con Paso 1 Hacer
		Escribir "[",[i],"] = ", datos[i];
	FinPara
FinSubProceso

SubProceso numPrimos()
	Definir x,num,contador Como Entero;
	Para num <- 1 Hasta 1000 Con Paso 1 Hacer
		x <- 1;
		contador <- 0;
		Mientras x <= num Hacer
			Si num mod x == 0 Entonces
				contador <- contador + 1;
			FinSi
			x <- x+ 1;
		FinMientras
		
		Si contador == 2 Entonces
			Escribir "El numero ", num, " Es Primo";
			
		FinSi
	FinPara
	
FinSubProceso

Proceso funcion_arreglos_3
	numPrimos();
FinProceso

		Escribir " ";
		
		
	FinPara
